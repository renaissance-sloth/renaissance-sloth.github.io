latex을 연습해보자.

<input id="latexInput"
       type="text"
       placeholder="Enter LaTeX code">

<div id="latexBox"></div>

<script>
document.addEventListener("DOMContentLoaded", () => {
  const input = document.getElementById("latexInput");
  const box = document.getElementById("latexBox");

  function render() {
    try {
      katex.render(
        input.value,
        box,
        {
          displayMode: false,
          throwOnError: false
        }
      );
    } catch (e) {
      box.textContent = e.message;
    }
  }

  input.addEventListener("input", render);

  render();
});
</script>
