<script>
var latexList = [
  "x^2 + 3x + 2 = 0",
  "\\int_0^1 e^{-x^2} dx",
  "\\sum_{n=1}^{\\infty} \\frac{1}{n^2} = \\frac{\\pi^2}{6}",
];

document.getElementById('latexBtn').onclick = function() {
  var idx = Math.floor(Math.random() * latexList.length);

  katex.render(
    latexList[idx],
    document.getElementById('latexBox'),
    {
      displayMode: true
    }
  );
};
</script>
